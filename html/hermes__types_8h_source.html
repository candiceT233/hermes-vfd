78 class PlacementPolicyConv {
79  public:
80  static std::string str(PlacementPolicy policy) {
81  switch (policy) {
82  case PlacementPolicy::kRandom: {
83  return "PlacementPolicy::kRandom";
84  }
85  case PlacementPolicy::kRoundRobin: {
86  return "PlacementPolicy::kRoundRobin";
87  }
88  case PlacementPolicy::kMinimizeIoTime: {
89  return "PlacementPolicy::kMinimizeIoTime";
90  }
91  case PlacementPolicy::kNone: {
92  return "PlacementPolicy::kNone";
93  }
94  }
95  return "PlacementPolicy::Invalid";
96  }
97 
98  static PlacementPolicy to_enum(const std::string &policy) {
99  if (policy.find("kRandom") != std::string::npos) {
100  return PlacementPolicy::kRandom;
101  } else if (policy.find("kRoundRobin") != std::string::npos) {
102  return PlacementPolicy::kRoundRobin;
103  } else if (policy.find("kMinimizeIoTime") != std::string::npos) {
104  return PlacementPolicy::kMinimizeIoTime;
105  } else if (policy.find("kNone") != std::string::npos) {
106  return PlacementPolicy::kNone;
107  }
108  return PlacementPolicy::kNone;
109  }
110 };
